func largeGroupPositions(S string) [][]int {
    min := 0
    res := make([][]int, 0)
    for i,_ := range S {
        if i == len(S) - 1 || S[i] != S[i+1] {
            if i - min + 1 >= 3 {
                res = append(res, []int{min, i})
            }
            min = i + 1
        }
    }
    return res
}
